    【axios】简单封装

    谦逊 5/12/2022 axios封装

    # axios封装案例

    # 思路

    封装axios
1.传递四个参数(url、method、params、data)   
2.返回一个promise对象
3.将params转换为query参数、method转换为大写
4.创建xhr以及判断状态码
5.resolve、reject分别返回response和error参数
6.最后给指定的GET/POST/PUT/DELETE函数添加返回成功函数和失败函数

    # 前言结构

        // 1. GET请求
    function GET() {
        axios({
            url: 'http://127.0.0.1:3000posts',
            method: 'GET',
            // 请求title为json-server的据
            // 这里只能通过id来查询数据
            params: {
                id: '1',
                title: 'json-server1',
                author: 'typicode1'
            }
        }).then(
            response => {
                console.log(response);
            },

                alert(error.message)
            }
        )
    }

    # 封装axios

        function axios({
        url,
        // 可以默认为GET
        method = 'GET',
        params = {},
        data = {}
    }) {
        // 返回一个promise对象
        return new Promise((resolve, reject) => {
            // 处理method小写问题
            method = method.toUpperCase()
            // 处理query参数  => 将params对象转换为?xxx=xx&xx=x
            /* 
             params: {
                id: '1',
                xxx: '123'
            }*/
            let queryString = params
            let result = ''
            // 将键名和键值拼接
            for (const [key, value] of Object.entries(queryString)) {
                let String = `${key}=${value}&`
                result += String
            }
            if (result) {
                result = result.slice(0, result.length - 1)
                url += '?' + result;
            }
            // 创建xhr
            const xhr = new XMLHttpRequest();
            // 初始化xhr
            xhr.open(method, url, true);
            // 判断请求方式
            if (method === 'GET' || method === 'DELETE') {
                xhr.send()
            } else if (method === 'POST' || method === 'PUT') {
                xhr.setRequestHeader('Content-Type', 'application/json;charset=utf-8');
                // 将需要发送的数据转换为JSON格式
                xhr.send(JSON.stringify(data))
            }
            // 绑定监听状态事件
            xhr.onreadystatechange = function () {
                // 判断请求状态码
                if (xhr.readyState === 4) {
                    const { status, statusText } = xhr
                    // 如果请求成功,就resolve一个response对象
                    if (status >= 200 && status < 300) {
                        // 准备结果对象
                        const response = {
                            // 这里必须转换为对象形式
                            data: JSON.parse(xhr.response),
                            status,
                            statusText
                        }
                        resolve(response)
                    } else {
                        // 接受一个Error对象
                        reject(new Error('请求失败!状态码为:' + xhr.status))
                    }
                }
            }
        })
    }
