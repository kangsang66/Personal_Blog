    【正则 / RegExp】总结

    谦逊 5/8/2022 正则
       /* 正则表达式作用:
    检查用户输入的格式是否符合预期
    正则表达式用于定义一些字符串的规则
    - 计算机可以根据正则表达式,来检查一个字符串是否符合规则
    - 或者将字符串中符合规则的内容提取出来
    创建正则表达式的对象
    语法:var 变量 = new RegExp('正则表达式','匹配模式')
    用来检查是否含有a  区别大小写
    在构造函数中可以传递一个匹配模式作为第二个参数
             可以是
                 i 忽略大小写
                 g 全局匹配模式
   */
   var reg = new RegExp('a', 'i');
   console.log(typeof reg); // object
   var str = 'Abc';
   // 正则表达式的方法:
   // - test()
   // 使用这个方法可以用来检查一个字符串是否符合正则表达式的规则
   console.log(reg.test(str)); // true 符合

   // 语法
   // 字面量创建:
   // var 变量 = /正则表达式/匹配模式
   // 字面量创建正则表达式更加简单 
   // 构造函数创建更加灵活 new RegExp()
   var reg = /a/i;
   console.log(reg.test('abc'));
   // 创建一个正则表达式,检测一个字符串中是否有a或b
   // 使用 | 表示或者的意思
   reg = /a|b/;
   console.log(reg.test('ac')); // true
   console.log(reg.test('bc')); // true
   // 创建一个正则表达式检查一个字符是否有字母
   // [] 里的内容也表示或的关系
   // [ab] == a|b
   // [a-z]  任意小写字母 中间的-表示从什么到什么
   // [A-Z]  任意小写字母 
   // [A-z]  任意字母
   reg = /[a-z]/;
   console.log(reg.test('z'));
   console.log(reg.test('Z'));
   // 创建一个正则表达式,检查一个字符串是否含有abc 或 adc 或aec
   reg = /a[bde]c/;
   console.log(reg.test('abc')); // true
   console.log(reg.test('acc')); // false
   // [^ ] 查找任何不在[]之间的字符
   reg = /[^ab]/
   console.log(reg.test('abc')); // true

   // 利用正则表达式来拆分一个字符串
   var str = '1a2b3c4d5e6f7';
   // let result = str.split(/[0-9]/)
   let result = str.split(/[A-z]/)  // 根据任意字母拆分
   console.log(result);
   /* search() 可以搜索字符串中是否含有指定内容
       类似indexOf() */
   str = 'hello abc hello acc adc aec';
   // 正则表达式不需要加引号
   console.log(str.search(/a[b-e]c/));
   /* match() 可以根据正则表达式,从一个字符串中将符合条件的内容提取出来
       - 默认情况下match() 只会找到第一个而不会全部查找
       - 需要设为g 全局匹配模式    
       - 可以为一个正则表达式设置多个匹配模式
   */
   str = 'H Io adbKlTbK';
   var a = str.match(/[A-Z]/g);
   var b = str.match(/[A-Z]/gi); // 忽略大小写
   console.log(a);
   /*
       replace() 可以根据正则表达式,替换指定内容
       - 
   */
   str = '1a2b3c4d5e6f7gsdasdADASD SDSA ASD SADsdasd231321';
   console.log(str.replace(/[A-z- ]/gi, '')); // 将字母删除

       /*
       创建一个正则表达式检查一个字符串是否含有aaa
   */
   /*
      量词
      - 通过量词可以设置一个内容出现的次数
      - 量词只对前面一个内容起作用
      - a{n}  a正好出现n次
      - (ab){n}  ab正好出现n次
      - a{m,n} 出现a m到n次
      - a{m,} 出现a m次以上
      - + 表示至少出现一次,相当于{1,}
      - * 0个或多个,相当于{0,} 
      - ? 0个或1个,相当于{0,1}
  */
   var reg = /aaa/;
   reg = /a{3}/; // 出现3次a
   console.log(reg.test('aaabbb'));
   reg = /(ab){3}/; // 出现3次ab
   console.log(reg.test('ababab'));
   reg = /ab{3}c/; // 出现3次b
   console.log(reg.test('abbbc111'));
   reg = /ab{1,3}c/; // 出现1次到3次b
   console.log(reg.test('abbc'));
   reg = /a+/;
   console.log(reg.test('aaa'));
   reg = /a*/
   console.log(reg.test('aa'));
   reg = /a?c/
   console.log(reg.test('ac'));
   /*
       检查一个字符串是否以a开头
       -  ^表示开头
       -  $表示结尾
   */
   reg = /^a/; // 匹配开头a
   console.log(reg.test('abc'));
   reg = /b$/ // 匹配结尾b
   console.log(reg.test('addddb'));
   reg = /^a$/ // 
   console.log(reg.test('a'));
   reg = /^a|a$/
   console.log(reg.test('abbb'));
   reg = /^a/
   console.log(reg.test('aaa'));
   /*
       案例:创建一个正则表达式判断输入是否是一个合法手机号
       手机号码规则:
       - 1. 一共11位数
       - 2. 第一位以1开头         ^1
       - 3. 第二位3-9任意数字     [3-9]
       - 4. 第三位以后9个任意数字  [0-9]{9}$
       */
   var phoneStr = '19138564021';
   var phoneReg = /^1[3-9][0-9]{9}$/
   console.log(phoneReg.test(phoneStr));

       /*
       创建一个正则表达式检查是否含有 '.'
       -   .表示任意字母
       -   在正则表达式中使用\作为转义字符
       -   用\.表示单纯的.
       -   在字符串中一个\必须用\\表示,因为会把后面的字符转义
   */
   var reg = /\./;
   reg = /\\/;
   reg = new RegExp('\\\\');
   console.log(reg.test('1\\'));
   console.log('\\\\'); // 输出\\
   console.log('\\'); // 输出\
   /*
       * \w
         - 任意字母、数字、下划线 [A-z0-9_]
       * \W
         - 除了字母、数字、下划线 [^A-z0-9_]
       * \d
         - 任意数字[0-9]
       * \D
         - 除了数字[0-9]
       * \s
         - 空格
       * \S
         - 除了空格
       * \b
         - 单词边界
       * \B
         - 除了单词边界
   */
   reg = /\w/;
   console.log(reg.test('asd123')); //true
   reg = /\W/;
   console.log(reg.test('*')); //true
   reg = /\d/;
   console.log(reg.test('123')); //true
   reg = /\D/;
   console.log(reg.test('abc')); //true
   reg = /\s/;
   console.log(reg.test(' 123')); //true
   reg = /\S/;
   console.log(reg.test('123')); //true
   /*
       用正则表达式检查一个字符串中是否含有单词child
   */
   //    
   reg = /\bchild\b/; //查找单词的单词child
   console.log(reg.test('child person')); //true
   reg = /\Bchild\B/;
   console.log(reg.test('childs')); //false
   /*
       接受一个用户的输入
       去除前后空格,保留用户输入之间的空格
       - 1.以多个空格开头 ^\s*
       - 2.以多个空格结尾 \s*$
       - 3.全局匹配  g
   */
   var str = '       ad—_   123min         ';
   console.log(str.replace(/^\s+|\s+$/g, ''));

       /*
       电子邮件
       - hello.li@abc.com
       格式:
       - 1. 任意字母、数字、下划线    \w{3,}
       - 2. .任意字母、数字、下划线(可有可无)  (\.\w+)*
       - 3. @任意字母、数字        @[A-z0-9]+
       - 4. .任意字母{2,5}       .[A-z]{2,5}
       - 5. .任意字母{2,5}       .[A-z]{2,5}
       */
   var emailReg = /^(\w{3,})(\.\w+)*@([A-z0-9])+((\.[A-z]{2,5}){1,2})$/;
   var str = '12311@163aad.com'
   console.log(str);
   console.log(emailReg.test(str));
